\newcommand{\classDate}[1]{
\tikzset{external/export next=false}
\begin{tikzpicture}[overlay,font=\tiny,align=left,
	xshift={-1.5em},yshift={-1em}]
	\node (date) [date] {};
	\node at (date.center) [text=black!70!white] {\scshape\bfseries #1};
\end{tikzpicture}
}
