%% TABLE OF STUFF
\usepackage[titles]{tocloft}
    \newlength{\figlength}
    % Insert word 'Abbildung' or 'Figure'
    \renewcommand{\cftfigpresnum}{\figurename\enspace}
    % Insert colon
    \renewcommand{\cftfigaftersnum}{:\enspace}
    % Start printing caption after the colon
    \settowidth{\figlength}{\cftfigpresnum\cftfigaftersnum}
    \addtolength{\cftfignumwidth}{\figlength}
    
    \newlength{\tablength}
    % Insert word 'Tabelle' or 'Table'
    \renewcommand{\cfttabpresnum}{\tablename\enspace}
    % Insert colon
    \renewcommand{\cfttabaftersnum}{:\enspace}
    % Start printing caption after the colon
    \settowidth{\tablength}{\cfttabpresnum\cfttabaftersnum}
    \addtolength{\cfttabnumwidth}{\tablength}
